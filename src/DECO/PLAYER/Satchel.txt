ACTOR Satchel : Weapon replaces RocketLauncher
{
	Weapon.BobStyle Alpha
	Weapon.BobSpeed 2.5
	Weapon.BobRangeX 0.2
	Weapon.BobRangeY 0.5
    weapon.ammotype1 "RocketAmmo"
    weapon.ammotype2 "ThrownCharge"
    weapon.ammouse1 1
    weapon.ammouse2 1
	scale 0.3
    weapon.ammogive 1
	Weapon.UpSound "weapons/putaway2"
	Weapon.SelectionOrder 5
	+WEAPON.NOAUTOAIM 
	+WEAPON.AMMO_CHECKBOTH
	+WEAPON.NOALERT
	States
	{
	Spawn:
		SATP A -1
		Stop
	Nothing:
		TNT1 A 1 A_WeaponReady(WRF_NOFIRE)
		goto Ready
	Select:
		TNT1 A 0 Offset(0, 32)
		TNT1 A 0 A_JumpIfInventory("ThrownCharge",1,"DetonatorDraw")
		TNT1 A 0 A_JumpIfInventory("RocketAmmo", 1, "ChargeDraw")
		goto Nothing
	Deselect: 
		  TNT1 AAAAAAAAAAAAAAAA 0 A_Lower
		  TNT1 AAAAAAA 1 A_Lower
		  Loop
	ChargeDraw:
		SACD A 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACD B 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACD C 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACD D 1 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACD E 1 
		SACD F 1 A_WeaponReady
		SACD G 1 A_WeaponReady
		SACG A 1 A_WeaponReady
		goto Ready
	ChargeDrawWith:
		TNT1 A 0
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACW A 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACW B 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACW C 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACW D 1 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACW E 1 
		SACW F 1 A_WeaponReady
		SACW G 1 A_WeaponReady
		SABG A 1 A_WeaponReady
		goto Ready
	DetonatorDraw:
		TNT1 A 0 A_JumpIfInventory("RocketAmmo", 1, "DrawBoth")
		SADD A 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SADD B 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SADD C 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SADD D 1 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SADD E 1 
		SADD F 1 A_WeaponReady
		SADD G 1 A_WeaponReady
		SADG A 1 A_WeaponReady
		goto Ready
	DrawBoth:
		SABD A 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SABD B 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SABD C 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SABD D 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SABD E 1 
		SABD F 1 A_WeaponReady
		SABD G 1 A_WeaponReady
		SABG A 1 A_WeaponReady
		goto Ready
	Ready:
		TNT1 A 0 A_JumpIfInventory("ThrownCharge",1,"ReadyDetonator")
		TNT1 A 0 A_JumpIfInventory("RocketAmmo",1,"ReadyCharge")
		Goto Nothing
	ReadyBoth:
		SABG A 1 A_WeaponReady
		Loop
	ReadyDetonator:
		TNT1 A 0 A_JumpIfInventory("RocketAmmo",1,"ReadyBoth")
		SADG A 1 A_WeaponReady
		Loop
	ReadyCharge:
		SACG A 1 A_WeaponReady
		Loop
	Fire:
		TNT1 A 0 A_JumpIfInventory("RocketAmmo",1,1)
		Goto Ready
		TNT1 A 0 A_JumpIfInventory("ThrownCharge",1,"FireNext")
		TNT1 A 0 A_FireCustomMissile("Charge",0,1,0,0)
		TNT1 A 0 A_GiveInventory("ThrownCharge",1)
		SATA A 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATA D 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATA E 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATA F 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATA H 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATA I 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A 1
		goto DetonatorDraw
	FireNext:
		TNT1 A 0 A_GiveInventory("ThrownCharge",1)
		SATG A 0 A_FireCustomMissile("Charge",0,1,0,0)
		SATW A 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATW D 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATW E 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATW F 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATW H 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SATW I 1
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SADG A 1
		TNT1 A 0 A_JumpIfInventory("RocketAmmo", 1, "ChargeDrawWith")
		goto Ready
	Altfire:
	TNT1 A 0 
	TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_JumpIfInventory("ThrownCharge",1,"AltFireDetonator")
		goto Ready
	AltFireBoth:
		SAXW A 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXW B 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXW C 3 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXW D 5 Thing_Destroy(1999)
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXW E 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXW F 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXW G 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SACG A 2
 		Goto Ready
	AltFireDetonator:
		TNT1 A 0 A_TakeInventory("ThrownCharge",0)
		TNT1 A 0 A_JumpIfInventory("RocketAmmo",1,"AltFireBoth")
		SAXA A 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXA B 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXA C 3 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXA D 5 Thing_Destroy(1999)
		SAXA E 2 
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXA F 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAXA G 2
		TNT1 A 0 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A 2
		Goto Ready
	}
}
